When a user clicks on a tab, the onSleep functions associated with all the tabs that will be hidden is called. If any of the onSleep functions return false, the click event is cancelled.





		
		<script>
			$('#demoSleeper').zTabs('property',{onsleep: function() {
					if($('#noSleep').attr('checked')) {
						return true;
					} else {
						alert('You must check the box before leaving this tab.');
						return false;
					}
				}
			});
		</script>
